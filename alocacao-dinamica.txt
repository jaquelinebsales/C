#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define MAX 10
int somarPares(int *ponteiro, int q_bytes);
int main(){

    int *p;
    int qtd;

    puts("Entre com a quantidade: ");
    scanf("%d", &qtd);

    p = malloc(qtd * sizeof(int));
    if(!p){
        puts("Memória insuficiente");
        exit(1);
    }

    srand(time(NULL));
    for(int i = 0; i < qtd ; i++){
        p[i] = rand()%MAX;
    }

    puts("Array gerado: ");
    for(int i = 0; i <qtd; i++){
        printf("%d \n", p[i]);
    }

    printf("A soma dos números pares do array gerado é: %d\n", somarPares(p, qtd));

    free(p);
    return 0;
}

int somarPares(int *ponteiro, int q_bytes){
    int soma = 0;
    for(int i = 0; i < q_bytes ; i++){
        if ( ponteiro[i] % 2 == 0){
            soma += ponteiro[i];
        }
    }

    return soma;
}

#include <stdlib.h>

#define MAX 100

int somarValoresMatriz(int *, int, int, int, int, int);

int main(){
    int linhas, colunas, *matriz = NULL, linha_usuario1, coluna_usuario1, linha_usuario2, coluna_usuario2;

    if(!matriz){
        puts("Memória insuficiente");
        exit(1);
    }

    srand(time(NULL));
    for(int i = 0; i < (linhas*colunas); i++){
        *(matriz+i) = rand() % MAX;
    }

    for(int  i = 0; i < (linhas*colunas); i++){
        printf("%5d", matriz[i]);
        if((i+1)%colunas==0){
            puts(" ");
        }
    }

    puts("Digite a linha e coluna do primeiro valor: ");
    scanf("%d %d", &linha_usuario1, &coluna_usuario1);

    puts("Digite a linha e coluna do secundo valor: ");

    if(linha_usuario1 <= linhas-1 && linha_usuario2 <= linhas-1 && coluna_usuario1<= colunas-1 && coluna_usuario2 <= colunas-1){
        printf("A soma dos valores é: %d \n", somarValoresMatriz(matriz, linha_usuario1, coluna_usuario1, linha_usuario2, coluna_usuario2, colunas));
    }else{
        puts("linha e/ou coluna inválida.");
        exit(1);
    }

    free(matriz);
    return 0;
}

int somarValoresMatriz(int *Matriz, int l_usuario1, int c_usuario1, int l_usuario2, int c_usuario2, int c_matriz){
    int soma = Matriz[c_matriz*(l_usuario1+1) - (c_matriz-c_usuario1)] + Matriz[c_matriz*(l_usuario2+1) - (c_matriz-c_usuario2)];
    return soma;
}
